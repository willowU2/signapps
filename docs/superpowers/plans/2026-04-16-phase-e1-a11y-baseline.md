- [ ] **Step 1: Write the catalog generator script**

Create `client/scripts/a11y-catalog.mjs`. The script uses `spawnSync` (no shell, no injection risk) to invoke eslint with JSON format, then aggregates.

```js
#!/usr/bin/env node
/**
 * A11y Catalog Generator — Phase E1a
 *
 * Usage: node scripts/a11y-catalog.mjs
 *
 * Reads the output of `npx eslint src/ --format json` and produces
 *   docs/bug-sweep/a11y-lint-catalog.md
 * with violations grouped by:
 *   1. Top rules by count (most impactful to fix first)
 *   2. Top files by count (hot spots)
 *   3. Full rule × file matrix for drilling
 *
 * Fails with exit 1 if eslint output is not valid JSON.
 */
import { spawnSync } from "node:child_process";
import { writeFileSync } from "node:fs";
import path from "node:path";

const SCRIPT_DIR = path.dirname(new URL(import.meta.url).pathname);
// On Windows, path.dirname(new URL(...).pathname) returns "/C:/Prog/..." — strip the leading "/"
const CLIENT_DIR = path.resolve(
  SCRIPT_DIR.startsWith("/") && /^\/[A-Za-z]:/.test(SCRIPT_DIR)
    ? SCRIPT_DIR.slice(1)
    : SCRIPT_DIR,
  "..",
);
const PROJECT_ROOT = path.resolve(CLIENT_DIR, "..");
const OUT_MD = path.join(PROJECT_ROOT, "docs", "bug-sweep", "a11y-lint-catalog.md");

console.log("Running eslint with JSON reporter (this may take 30–60 s)…");

// Use spawnSync with explicit argv (no shell). Windows needs `npx.cmd`.
const isWin = process.platform === "win32";
const result = spawnSync(
  isWin ? "npx.cmd" : "npx",
  ["eslint", "src/", "--format", "json", "--no-error-on-unmatched-pattern"],
  {
    cwd: CLIENT_DIR,
    encoding: "utf-8",
    maxBuffer: 100 * 1024 * 1024,
  },
);

const raw = result.stdout || "";
if (!raw.trim().startsWith("[")) {
  console.error("ESLint did not return JSON. stderr:");
  console.error(result.stderr?.slice(0, 1000) || "(empty)");
  console.error("stdout head:", raw.slice(0, 200));
  process.exit(1);
}

const results = JSON.parse(raw);

// Flatten: [{ filePath, messages: [{ ruleId, severity, ... }] }] → a11y messages only
const a11yMessages = [];
for (const fileResult of results) {
  for (const msg of fileResult.messages) {
    if (msg.ruleId && msg.ruleId.startsWith("jsx-a11y/")) {
      a11yMessages.push({
        file: path
          .relative(PROJECT_ROOT, fileResult.filePath)
          .replace(/\\/g, "/"),
        rule: msg.ruleId,
        line: msg.line,
        column: msg.column,
        message: msg.message,
      });
    }
  }
}

// Group by rule
const byRule = new Map();
for (const m of a11yMessages) {
  byRule.set(m.rule, (byRule.get(m.rule) || 0) + 1);
}
const ruleEntries = [...byRule.entries()].sort((a, b) => b[1] - a[1]);

// Group by file
const byFile = new Map();
for (const m of a11yMessages) {
  byFile.set(m.file, (byFile.get(m.file) || 0) + 1);
}
const fileEntries = [...byFile.entries()].sort((a, b) => b[1] - a[1]);

// Rule × file pairs
const byRuleFile = new Map();
for (const m of a11yMessages) {
  const k = `${m.rule}::${m.file}`;
  byRuleFile.set(k, (byRuleFile.get(k) || 0) + 1);
}

// Build markdown
const now = new Date().toISOString().slice(0, 10);
const lines = [];
lines.push(`# A11y Lint Catalog — ${now}`);
lines.push("");
lines.push(`Source: \`eslint-plugin-jsx-a11y\` on \`client/src/**/*.{ts,tsx}\``);
lines.push(`Generated by: \`client/scripts/a11y-catalog.mjs\``);
lines.push("");
lines.push(`**Total jsx-a11y warnings:** ${a11yMessages.length}`);
lines.push(`**Distinct rules:** ${ruleEntries.length}`);
lines.push(`**Distinct files affected:** ${fileEntries.length}`);
lines.push("");

lines.push("## Top rules by count");
lines.push("");
lines.push("| Rule | Count | Top 3 files |");
lines.push("|---|---|---|");
for (const [rule, count] of ruleEntries) {
  const topFiles = [...byRuleFile.entries()]
    .filter(([k]) => k.startsWith(`${rule}::`))
    .sort((a, b) => b[1] - a[1])
    .slice(0, 3)
    .map(([k, c]) => `\`${k.split("::")[1]}\` (${c})`)
    .join(", ");
  lines.push(`| \`${rule}\` | ${count} | ${topFiles} |`);
}
lines.push("");

lines.push("## Top 30 files by count");
lines.push("");
lines.push("| File | Count |");
lines.push("|---|---|");
for (const [file, count] of fileEntries.slice(0, 30)) {
  lines.push(`| \`${file}\` | ${count} |`);
}
lines.push("");

lines.push("## Prioritization guidance for E1c");
lines.push("");
lines.push(
  "Apply fixes in the order of `Top rules by count`. For each rule:",
);
lines.push(
  "1. Check if `eslint --fix` resolves it automatically (some are auto-fixable).",
);
lines.push(
  "2. If not, check if the top-3 files are shared components (`components/ui/*`): a single fix there propagates.",
);
lines.push("3. Otherwise, decide: individual fixes or defer to Phase E2.");
lines.push("");

writeFileSync(OUT_MD, lines.join("\n"), "utf-8");
console.log(`Wrote ${OUT_MD}`);
console.log(
  `Total: ${a11yMessages.length} warnings, ${ruleEntries.length} rules, ${fileEntries.length} files.`,
);
```

```ts
/**
 * A11y Audit Spec — Phase E1b
 *
 * Iterates the full list of authenticated routes + a handful of public
 * routes, runs axe-core on each, serializes violations to
 *   docs/bug-sweep/a11y-axe-baseline.json
 *
 * This spec does NOT fail on violations — it only collects the baseline.
 * Phase E1c fixes reduce the count; the test can be promoted to a gate
 * once stable.
 *
 * Prerequisites:
 *   - All backend services running
 *   - Dev server on http://localhost:3000
 *   - Authenticated storage state (auth.setup.ts)
 *
 * Run:
 *   cd client && npx playwright test e2e/a11y-audit.spec.ts --project=chromium
 */
import { test } from "@playwright/test";
import AxeBuilder from "@axe-core/playwright";
import { writeFileSync, mkdirSync } from "node:fs";
import path from "node:path";

// ── Route inventory ─────────────────────────────────────────────────────────

const STATIC_ROUTES: string[] = [
  // REPLACE THIS ARRAY with the output of Step 1 of this task.
  // Representative subset for initial testing:
  "/",
  "/login",
  "/forms",
  "/mail",
  "/chat",
  "/docs",
  "/cal",
  "/storage",
  "/admin",
  "/vault",
  "/accounting",
  "/contacts",
  "/crm",
  "/workforce",
  "/projects",
  "/notifications",
  "/meet",
  "/remote",
  "/pxe",
  "/tasks",
  "/social",
  "/metrics",
  "/ai",
];

/**
 * Parameterized routes with seed IDs. IDs come from the seed fixtures
 * (see tools/signapps-seed). If the seed is absent, these will hit 404s;
 * the audit still catches layout-level a11y issues.
 */
const PARAM_ROUTES: string[] = [
  "/f/00000000-0000-0000-0000-000000000101",
  "/forms/00000000-0000-0000-0000-000000000101",
];

const ALL_ROUTES = [...STATIC_ROUTES, ...PARAM_ROUTES];

// ── The audit ───────────────────────────────────────────────────────────────

interface RouteResult {
  route: string;
  status: "ok" | "error";
  violations: Array<{
    id: string;
    impact: string | undefined;
    help: string;
    helpUrl: string;
    nodes: number;
  }>;
  error?: string;
  timing_ms: number;
}

test.describe("A11y baseline audit", () => {
  test.describe.configure({ mode: "serial" });

  test("audit all routes with axe-core", async ({ page }) => {
    // ~30 min for 270 routes. Generous timeout.
    test.setTimeout(45 * 60 * 1000);

    const results: RouteResult[] = [];

    for (const route of ALL_ROUTES) {
      const started = Date.now();
      try {
        await page.goto(`http://localhost:3000${route}`, {
          waitUntil: "domcontentloaded",
          timeout: 30_000,
        });

        // Wait for real content before auditing (axe on an empty SPA
        // shell produces noise).
        await page
          .locator("main, nav, [data-slot='sidebar'], form, [data-page='login']")
          .first()
          .waitFor({ state: "visible", timeout: 10_000 })
          .catch(() => {});

        const { violations } = await new AxeBuilder({ page })
          .disableRules([
            // color-contrast audit is a dedicated session (Tailwind v4
            // tokens). Too noisy at the design-system level.
            "color-contrast",
          ])
          .analyze();

        results.push({
          route,
          status: "ok",
          violations: violations.map((v) => ({
            id: v.id,
            impact: v.impact,
            help: v.help,
            helpUrl: v.helpUrl,
            nodes: v.nodes.length,
          })),
          timing_ms: Date.now() - started,
        });
        console.log(
          `[${results.length}/${ALL_ROUTES.length}] ${route} — ${violations.length} violations (${Date.now() - started}ms)`,
        );
      } catch (err) {
        results.push({
          route,
          status: "error",
          violations: [],
          error: err instanceof Error ? err.message : String(err),
          timing_ms: Date.now() - started,
        });
        console.warn(
          `[${results.length}/${ALL_ROUTES.length}] ${route} — ERROR: ${err}`,
        );
      }
    }

    const outDir = path.resolve(__dirname, "..", "..", "docs", "bug-sweep");
    mkdirSync(outDir, { recursive: true });
    const outPath = path.join(outDir, "a11y-axe-baseline.json");
    writeFileSync(
      outPath,
      JSON.stringify(
        {
          generated_at: new Date().toISOString(),
          total_routes: ALL_ROUTES.length,
          results,
        },
        null,
        2,
      ),
      "utf-8",
    );

    const totalViolations = results.reduce(
      (sum, r) => sum + r.violations.length,
      0,
    );
    console.log(
      `\n=== a11y audit complete ===\nRoutes: ${results.length}\nTotal violations: ${totalViolations}\nErrors: ${results.filter((r) => r.status === "error").length}\nOutput: ${outPath}\n`,
    );
  });
});
```

Append to `client/scripts/a11y-catalog.mjs` (before the final `console.log`):

```js
// ── Axe baseline summary ────────────────────────────────────────────────────

import { readFileSync, existsSync } from "node:fs";

const AXE_JSON = path.join(
  PROJECT_ROOT,
  "docs",
  "bug-sweep",
  "a11y-axe-baseline.json",
);
const AXE_MD = path.join(
  PROJECT_ROOT,
  "docs",
  "bug-sweep",
  "a11y-axe-summary.md",
);

if (!existsSync(AXE_JSON)) {
  console.log(`Axe baseline not found at ${AXE_JSON} — skipping summary.`);
  console.log(
    "Run the audit first: cd client && npx playwright test e2e/a11y-audit.spec.ts",
  );
  process.exit(0);
}

const axeData = JSON.parse(readFileSync(AXE_JSON, "utf-8"));

// Group by rule across all routes
const axeByRule = new Map();
for (const routeResult of axeData.results) {
  for (const v of routeResult.violations || []) {
    const cur = axeByRule.get(v.id) || {
      count: 0,
      impact: v.impact,
      help: v.help,
      routes: new Set(),
    };
    cur.count += v.nodes;
    cur.routes.add(routeResult.route);
    axeByRule.set(v.id, cur);
  }
}

// Group by impact
const byImpact = {
  critical: 0,
  serious: 0,
  moderate: 0,
  minor: 0,
  unknown: 0,
};
for (const routeResult of axeData.results) {
  for (const v of routeResult.violations || []) {
    const key = v.impact || "unknown";
    byImpact[key] = (byImpact[key] || 0) + v.nodes;
  }
}

const axeLines = [];
axeLines.push(
```
